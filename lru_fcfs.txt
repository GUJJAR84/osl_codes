#include <iostream>
using namespace std;

void fcfs(int pages[], int n, int frameSize) {
    int frame[10], index = 0, faults = 0;

    for(int i=0;i<frameSize;i++)
        frame[i] = -1;

    cout << "\nFCFS Page Replacement:\n";

    for(int i=0;i<n;i++) {
        bool found = false;

        // Check if already in frame
        for(int j=0;j<frameSize;j++) {
            if(frame[j] == pages[i]) {
                found = true;
                break;
            }
        }

        if(!found) {
            frame[index] = pages[i];
            index = (index + 1) % frameSize;
            faults++;
        }

        cout << "After page " << pages[i] << " : ";
        for(int j=0;j<frameSize;j++)
            cout << frame[j] << " ";
        cout << "\n";
    }

    cout << "Total Page Faults (FCFS): " << faults << "\n";
}

// ------------------ LRU ------------------

int findLRU(int time[], int frameSize) {
    int min = time[0], pos = 0;
    for(int i=1;i<frameSize;i++) {
        if(time[i] < min) {
            min = time[i];
            pos = i;
        }
    }
    return pos;
}

void lru(int pages[], int n, int frameSize) {
    int frame[10], time[10];
    int faults = 0, counter = 0;

    for(int i=0;i<frameSize;i++)
        frame[i] = -1;

    cout << "\nLRU Page Replacement:\n";

    for(int i=0;i<n;i++) {
        bool found = false;

        // check if page exists
        for(int j=0;j<frameSize;j++) {
            if(frame[j] == pages[i]) {
                found = true;
                time[j] = ++counter;  
                break;
            }
        }

        if(!found) {
            int pos = -1;

            // find an empty frame
            for(int j=0;j<frameSize;j++) {
                if(frame[j] == -1) {
                    pos = j;
                    break;
                }
            }

            if(pos == -1)        // no empty frame â†’ replace LRU
                pos = findLRU(time, frameSize);

            frame[pos] = pages[i];
            time[pos] = ++counter;
            faults++;
        }

        cout << "After page " << pages[i] << " : ";
        for(int j=0;j<frameSize;j++)
            cout << frame[j] << " ";
        cout << "\n";
    }

    cout << "Total Page Faults (LRU): " << faults << "\n";
}

int main() {
    int pages[20], n = 17, frameSize;

    cout << "Enter 17 page requests:\n";
    for(int i=0;i<n;i++)
        cin >> pages[i];

    cout << "Enter frame size (min 3): ";
    cin >> frameSize;

    fcfs(pages, n, frameSize);
    lru(pages, n, frameSize);

    return 0;
}
